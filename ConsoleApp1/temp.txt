    class BubbleSort//class don't have signiture 
	{
        int temp;

        public void sort(int[] array)
        {
            for (int i = 0; i < array.Length - 1; i++)
            {
                for (int j = 0; j < array.Length - (1 + i); i++)
                {
                    if (array[j] > array[j + 1])
                    {
                        temp = array[j];
                        array[j] = array[j + 1];
                        array[j + 1] = temp;
                        //Console.WriteLine(array[j]);

                    }

                }
            }
        }
    }
